using System;

class PeakElement
{
    // Function to find peak element
    public static int FindPeak(int[] arr)
    {
        int low = 0;
        int high = arr.Length - 1;

        while (low <= high)
        {
            int mid = low + (high - low) / 2;

            // Check if mid is the peak element
            if ((mid == 0 || arr[mid - 1] <= arr[mid]) && 
                (mid == arr.Length - 1 || arr[mid + 1] <= arr[mid]))
            {
                return arr[mid];  // Found the peak element
            }
            // If the element at mid is smaller than the element at mid + 1, 
            // then the peak lies on the right side
            else if (mid < arr.Length - 1 && arr[mid] < arr[mid + 1])
            {
                low = mid + 1;
            }
            // If the element at mid is smaller than the element at mid - 1, 
            // then the peak lies on the left side
            else
            {
                high = mid - 1;
            }
        }

        return -1;  // This will never be reached if the array has at least one peak
    }

    static void Main(string[] args)
    {
        int[] arr = { 1, 3, 20, 4, 1, 0 };  // Example array
        int peak = FindPeak(arr);
        
        Console.WriteLine("Peak element is: " + peak);
    }
}
